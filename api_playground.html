BACKEND FIX · clean_html_stream

If your LLM streams thinking text before HTML, add this filter in main.py:

async def clean_html_stream(gemini_stream):
    import re
    html_started = False
    buffer = ""
    HTML_RE = re.compile(
        r'<!doctype\s+html|<html[\s>]',
        re.IGNORECASE
    )
    async for chunk in gemini_stream:
        text = chunk.content if hasattr(chunk,"content") else str(chunk)
        if not text:
            continue
        # Strip markdown fences
        text = text.replace("```html\n","").replace("```html","").replace("```","")
        if html_started:
            yield text
            continue
        buffer += text
        m = HTML_RE.search(buffer)
        if m:
            html_started = True
            yield buffer[m.start():]  # discard preamble, yield from HTML
            buffer = ""
